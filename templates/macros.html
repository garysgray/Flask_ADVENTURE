{# ===================== Map Cell Macros ===================== #} {# Returns the CSS class for a map cell based on player position and visited state #} {% macro cell_class(pos, player_pos, visited_rooms) %} {% if pos == player_pos %}player {% elif pos in visited_rooms %}room {% else %}hidden{% endif %} {% endmacro %} {# Returns stair icons if the room has up/down exits #} {% macro stair_icons(cell) %} {% if 'up' in cell.exits %}▲{% endif %} {% if 'down' in cell.exits %}▼{% endif %} {% endmacro %} {# Returns ✦ if the room has any cardinal direction teleport exits #} {% macro teleport_icon(cell) %} {% set has_teleport = namespace(found=false) %} {% for dir in cell.exit_destinations %} {% if dir in ['north', 'south', 'east', 'west'] %} {% set has_teleport.found = true %} {% endif %} {% endfor %} {% if has_teleport.found %}✦{% endif %} {% endmacro %} {# Renders the full contents of a single map cell #} {% macro cell_contents(pos, cell, player_pos, visited_rooms) %} {% set visible = (pos == player_pos or pos in visited_rooms) %} {% if pos == player_pos %}P{% endif %} {% if cell and visible %} {{ stair_icons(cell) }} {{ teleport_icon(cell) }} {% endif %} {% endmacro %}
